#### Code: skeleton service (FastAPI, Python)

```python
# app.py
from fastapi import FastAPI, Request
from pydantic import BaseModel
from llm import generate
from crm import upsert_contact, create_deal, update_deal, log_activity
from mailer import send_email
from scheduler import schedule_post

app = FastAPI()

class EmailEvent(BaseModel):
    from_addr: str
    name: str | None = None
    subject: str
    body: str
    thread_id: str | None = None

@app.post("/webhooks/email")
async def on_email(event: EmailEvent):
    contact = upsert_contact(email=event.from_addr, name=event.name or "")
    deal_id = create_deal(contact_id=contact.id, source="email", context={}).id

    analysis = generate(
        system="Classify: positive, neutral, objection, pricing, unsubscribe, ooo, bounce. "
               "Return JSON with {intent, score:0-100, summary, next_stage, next_action}.",
        user=event.body
    )

    if analysis["intent"] == "unsubscribe":
        log_activity(contact.id, deal_id, "email", "inbound", event.body, {"intent":"unsubscribe"})
        return {"ok": True}

    reply = generate(
        system="Write a concise, friendly sales reply under 120 words that advances the deal. "
               "Offer two meeting slots, ask one qualifying question.",
        user=event.body
    )

    send_email(to=event.from_addr, subject=f"Re: {event.subject}", body=reply)
    update_deal(deal_id, stage=analysis["next_stage"], score=analysis["score"],
                notes=analysis["summary"], next_action=analysis["next_action"])
    log_activity(contact.id, deal_id, "email", "outbound", reply, {"intent":analysis["intent"]})
    return {"ok": True}

class ChatMsg(BaseModel):
    session_id: str
    message: str

@app.post("/webhooks/chat")
async def on_chat(msg: ChatMsg):
    # Qualify quickly: pain, goal, timeframe, budget, contact
    bot = generate(
        system="You are ARGILETTE’s site assistant. Ask focused questions to qualify lead in <=4 turns. "
               "Be helpful; if qualified, offer booking link.",
        user=msg.message
    )
    # Persist transcript, create/update contact and deal
    return {"reply": bot}

@app.post("/cron/social/generate")
async def cron_social():
    brief = "ICP: B2B SaaS founders; Topics: CRM automation, AI outreach, deal acceleration."
    ideas = generate(system="Create 5 post ideas with hooks + CTA for LinkedIn, X, IG.", user=brief)
    for idea in ideas if isinstance(ideas, list) else [ideas]:
        post = generate(system="Rewrite for LinkedIn; add 2 relevant hashtags; 220-260 words max.", user=idea)
        post_id = schedule_post(platform="linkedin", content=post)
        # log to CRM
    return {"scheduled": True}
```

Placeholders: implement llm.generate, crm.*, mailer.*, scheduler.* to match your stack.
