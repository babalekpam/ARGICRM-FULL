typescriptrouter.get("/credits", authenticate, async (req, res) => {
  try {

        : 0,
    });

    console.error("GET /marketplace/credits error:", err);
    res.status(500).json({ error: "Failed to fetch credits" });
  }

    // Accept both naming conventions

                          `INV-${Date.now()}`;
    const total         = body.total     || body.amount        || 0;
    const clientName    = body.clientName|| body.client        || body.name || "";

      total:    total,

      createdAt: new Date(),

    res.status(201).json(invoice[0]);
