javascript// Cost calculator
function calculateCost(usage) {
  const inputCost  = (usage.input_tokens  / 1000000) * 3.00;
  const outputCost = (usage.output_tokens / 1000000) * 15.00;
  return inputCost + outputCost;
}

LAYER 2 — HARD USAGE LIMITS PER CLIENT
Before every Claude API call, check if the client has credits remaining:
javascriptasync function checkCreditsAndCall(tenantId, systemPrompt, message) {

  // 1. Check remaining credits
  const client = await db.query(`
    SELECT ai_credits_remaining, plan_type 
    FROM tenants WHERE id = $1
  `, [tenantId]);

  // 2. BLOCK if out of credits
  if (client.rows[0].ai_credits_remaining <= 0) {
    throw new Error("AI credits exhausted. Please upgrade your plan.");
  }

  // 3. Call Claude
  const response = await askClaude(systemPrompt, message);

  // 4. Deduct 1 credit
  await db.query(`
    UPDATE tenants 
    SET ai_credits_remaining = ai_credits_remaining - 1 
    WHERE id = $1
  `, [tenantId]);

  return response;
}
